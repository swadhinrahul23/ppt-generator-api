# Vercel Environment Variables for Salesforce Storage
# Add these in your Vercel dashboard under Settings > Environment Variables

# Storage Configuration
STORAGE_TYPE=salesforce

# Salesforce Credentials
SALESFORCE_CLIENT_ID=your_connected_app_consumer_key
SALESFORCE_CLIENT_SECRET=your_connected_app_consumer_secret
SALESFORCE_USERNAME=your_salesforce_username
SALESFORCE_PASSWORD=your_salesforce_password
SALESFORCE_SECURITY_TOKEN=your_salesforce_security_token
SALESFORCE_LOGIN_URL=https://login.salesforce.com

# For sandbox orgs, use: https://test.salesforce.com
# For production orgs, use: https://login.salesforce.com
